class robot() :
	def __init__(self,mv,av,pos=(0,0),dir=S):
		self.mv = mv
		self.av = av
		self.pos = pos
		self.dir = dir

	def move(self):
		for i in range(self.mv):
			move_forward()
			if get_sensors()[0][1]==black:
				move_backward(i+1)
				return False
		self.pos=add(self.pos,self.dir)
		return True
